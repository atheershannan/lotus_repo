  איפה מעדכנים את ה-URL של Railway Backend
═══════════════════════════════════════════════════════════

לאחר יצירת Service Domain ב-Railway, עדכן ב-3 מקומות:

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. VERCEL ENVIRONMENT VARIABLES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

מהיכן:
Vercel Dashboard → Your Project → Settings → Environment Variables

מה לעדכן:
REACT_APP_API_URL = https://your-new-railway-domain.up.railway.app/api

כיצד:
1. פתח את Vercel Dashboard
2. בחר את הפרויקט
3. Settings → Environment Variables
4. מצא REACT_APP_API_URL
5. לחץ Edit
6. עדכן את ה-URL החדש
7. Save
8. Redeploy

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
2. RAILWAY_URL.txt (במקרה של שינוי)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

קובץ: RAILWAY_URL.txt

עדכן את השורות:
🔗 Railway URL:
https://your-new-railway-domain.up.railway.app

🌐 API Endpoint:
https://your-new-railway-domain.up.railway.app/api

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
3. GITHUB SECRETS (אם משתמש ב-GitHub Actions)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

מהיכן:
GitHub → Your Repository → Settings → Secrets and variables → Actions

מה לעדכן:
